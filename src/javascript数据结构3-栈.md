后进先出（LIFO,last-in-first-out）的数据结构
类比：堆叠盘子，只能从上面拿走盘子

		function Stack() {
		   this.dataStore = [];
		   this.pos = 0;
		   this.push=push;
		   this.pop=pop;
		   this.peek=peek;
		   this.clear = clear;
		   this.length=length;
		}
        
        function push(element){
        	this.dataStore[this.pos++]=element;
        }
        function peek(){
        	return this.dataStore[this.top-1];
        }
        function pop(){
        	return this.dataStore[--this.top];
        }
        function clear(){
        	this.top=0;
        }
        function length(){
        	return this.top;
        }
       /************************************************************************/
